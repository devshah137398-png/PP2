#include <stdio.h>

struct Date {
    int day, month, year;
};

int main() {
    struct Date d1, d2;
    printf("Enter Date 1 (d m y): ");
    scanf("%d %d %d", &d1.day, &d1.month, &d1.year);
    printf("Enter Date 2 (d m y): ");
    scanf("%d %d %d", &d2.day, &d2.month, &d2.year);

    if(d1.day == d2.day && d1.month == d2.month && d1.year == d2.year)
        printf("equal");
    else
        printf("not equal");
    return 0;
}
