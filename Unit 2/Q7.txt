#include <stdio.h>

struct complex {
    float real;
    float imag;
};

struct complex add(struct complex n1, struct complex n2) {
    struct complex temp;
    temp.real = n1.real + n2.real;
    temp.imag = n1.imag + n2.imag;
    return temp;
}

int main() {
    struct complex c1 = {2.3, 4.5}, c2 = {1.2, 3.1}, sum;
    sum = add(c1, c2);
    printf("Sum = %.1f + %.1fi", sum.real, sum.imag);
    return 0;
}
