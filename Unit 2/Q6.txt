#include <stdio.h>

struct student {
    char name[20];
    int year_of_joining;
};

void print2019Joiners(struct student s[], int n) {
    for(int i = 0; i < n; i++) {
        if(s[i].year_of_joining == 2019) printf("Name: %s\n", s[i].name);
    }
}

int main() {
    struct student s[2] = {{"John", 2019}, {"Sara", 2020}};
    print2019Joiners(s, 2);
    return 0;
}
