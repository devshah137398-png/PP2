#include <stdio.h>
#include <string.h>

struct student {
    int rollno; char name[20]; char department[40]; int year_of_joining; float score;
};

int main() {
    struct student s[5], temp;
    // Assume data is already entered for brevity
    for(int i=0; i<5; i++) scanf("%s", s[i].name); 

    for(int i = 0; i < 4; i++) {
        for(int j = i + 1; j < 5; j++) {
            if(strcmp(s[i].name, s[j].name) > 0) {
                temp = s[i]; s[i] = s[j]; s[j] = temp;
            }
        }
    }
    return 0;
}
