#include <stdio.h>

struct student {
    char name[20];
    float score;
};

void printHighScorers(struct student s[], int n) {
    printf("Students with score > 5.0:\n");
    for(int i = 0; i < n; i++) {
        if(s[i].score > 5.0) printf("%s\n", s[i].name);
    }
}

int main() {
    struct student s[3] = {{"Alice", 6.2}, {"Bob", 4.5}, {"Charlie", 8.0}};
    printHighScorers(s, 3);
    return 0;
}
