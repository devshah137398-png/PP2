#include <stdio.h>

struct student {
    int rollno;
    char name[20];
    char department[40];
    int year_of_joining;
    float score;
};

int main() {
    struct student s;

    printf("Enter Roll No: ");
    scanf("%d", &s.rollno);
    printf("Enter Name: ");
    scanf("%s", s.name);
    printf("Enter Department: ");
    scanf("%s", s.department);
    printf("Enter Year of Joining: ");
    scanf("%d", &s.year_of_joining);
    printf("Enter Score: ");
    scanf("%f", &s.score);

    printf("\n--- Student Details ---\n");
    printf("Roll No: %d\nName: %s\nDept: %s\nYear: %d\nScore: %.2f\n", 
            s.rollno, s.name, s.department, s.year_of_joining, s.score);

    return 0;
}
